import pandas as pd

dict_obj={'type':[11,12,12,11,12,12,13],'sjl':[1,2,1,2,3,2,1]}

print(df_obj)
print(type(df_obj.groupby('type')))

grouptype1 = df_obj.groupby('type')
print(grouptype1)
print(grouptype1.sum())

print(grouptype2.sum())
