## Background

​	最近一个项目需要到Maxcompute中获取一些表名,表结构,表的数据量,各个字段的空值情况,以及可能的数据结构。



## Summary

- 利用pyodps提供`for table in odps.list_tables():`,`for column in t.schema.columns:`获取表的信息,字段信息
- 利用python将数据写入到postgresql数据库

## Question

- 原本要利用python写的脚本,使用postgresql数据库脚本来替代了



## Example

### 获取所有该模式下的表名,表结构

	print(table.name)
```

```

### 小结

```
#coding=utf-8
import os
import psycopg2
from odps import ODPS

odps = ODPS()

conn = psycopg2.connect(database="postgres", user="ysys", password="ysys", host="192.168.1.103", port="5432")
cur = conn.cursor()
cur2 = conn.cursor()
cur3 = conn.cursor()


## 每次执行清空临时表
cur.execute('truncate table odps_etl.tmp_list_tables')
cur.execute('truncate table odps_etl.tmp_list_columns')
## 读取schema gh_ysys 下的表
for table in odps.list_tables():
	t = odps.get_table(table.name)
	with odps.execute_sql('select count(*) as sjl from '+ table.name).open_reader() as reader:
		for record in reader:
			tablecount = record.sjl
			cur.execute("insert into odps_etl.tmp_list_tables(tablename,tablesize,tablecounts) values(%s,%s,%s)",(table.name,t.size,tablecount));
	for column in t.schema.columns:
		cur.execute("insert into odps_etl.tmp_list_columns(tablename,columntext) values(%s,%s)",(table.name,str(column)));
cur.execute('select odps_etl.f_update_tmp_list_tables()')
cur.execute('select odps_etl.f_update_tmp_list_columns()')

cur2.execute('select countsql from odps_etl.tmp_list_columns')
for row in cur2:
	row = str(row).replace("('",'')
	row = str(row).replace("',)",'')
	row = str(row).replace('("','')
	row = str(row).replace('",)','')
	print(row)
	with odps.execute_sql(row).open_reader() as readers:
		for records in readers:
			columncount = records.sjl
			cur.execute('update odps_etl.tmp_list_columns set countnotnull = %s where tablename = %s and columnname = %s ',(columncount,records.tbname,records.clname))

cur3.execute("select regsql from odps_etl.tmp_list_columns where columntype='string'")
for row3 in cur3:
	row3 = str(row3).replace("('",'')
	row3 = str(row3).replace("',)",'')
	row3 = str(row3).replace('("','')
	row3 = str(row3).replace('",)','')
	row3 = str(row3).replace("\\","\\")
	print(row3)			
	with odps.execute_sql(row3).open_reader() as readers3:
		for records3 in readers3:
			cur.execute('insert into odps_etl.tmp_table_sjjg(sjl,bm,zdm,zdz) values(%s,%s,%s,%s)',(records3.sjl,records3.tbname,records3.clname,records3.column_sjjgz))
conn.commit()
conn.close()

```
