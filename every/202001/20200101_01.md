from odps.udf import annotate
import re
import datetime
@annotate("string->string")
class dateVaild(object):
    def evaluate(self, var):
        if var is None:
            return '4'
        if len(var)!=8:
            return '1'
        if not re.match("^\d{8}$",var):
            return '2'
        try: 
            datetime.datetime.strptime(var,'%Y%m%d')
            return '0'
        except ValueError:
            return '3'
