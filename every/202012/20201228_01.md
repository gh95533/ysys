```
# wget http://li.nux.ro/download/nux/dextop/el7/x86_64/nux-dextop-release-0-5.el7.nux.noarch.rpm
# yum install -y epel-release
# rpm -ivh nux-dextop-release-0-5.el7.nux.noarch.rpm
# yum install webkitgtk
```
